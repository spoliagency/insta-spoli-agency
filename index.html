Conecte com Google Sheets

Crie uma planilha, abra Extensoes > Apps Script, cole o codigo ao lado e publique como aplicativo da Web. Depois cole a URL gerada na constante LEAD_WEBHOOK_URL dentro deste HTML.

No GitHub Pages, o HTML nao pode guardar token privado do Notion ou Google. O Apps Script roda na sua conta e recebe os leads sem expor senha no navegador. 
const SHEET_NAME = 'Leads';

function doPost(e) {
  const lock = LockService.getScriptLock();
  lock.waitLock(10000);

  try {
    const sheet = getLeadSheet_();
    const data = JSON.parse(e.postData.contents || '{}');

    sheet.appendRow([
      new Date(),
      data.name || '',
      data.phone || '',
      data.email || '',
      data.materialTitle || '',
      data.materialId || '',
      data.source || '',
      data.pageUrl || '',
      data.userAgent || ''
    ]);

    return ContentService
      .createTextOutput(JSON.stringify({ ok: true }))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (error) {
    return ContentService
      .createTextOutput(JSON.stringify({ ok: false, error: String(error) }))
      .setMimeType(ContentService.MimeType.JSON);
  } finally {
    lock.releaseLock();
  }
}

function getLeadSheet_() {
  const spreadsheet = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = spreadsheet.getSheetByName(SHEET_NAME);

  if (!sheet) {
    sheet = spreadsheet.insertSheet(SHEET_NAME);
  }

  if (sheet.getLastRow() === 0) {
    sheet.appendRow([
      'Data',
      'Nome',
      'Telefone',
      'Email',
      'Material',
      'ID do material',
      'Origem',
      'Pagina',
      'Navegador'
    ]);
  }

  return sheet;
}
